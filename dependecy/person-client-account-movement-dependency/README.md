Biblioteca compartida (DTOs, enums, constantes, integración gRPC y REST) para db-layer, core y composite. Paquete Java base: `com.personclientaccountmovement`.

<dependency>
    <groupId>com.personclientaccountmovement</groupId>
    <artifactId>person-client-account-movement-dependency</artifactId>
    <version>${version.person-client-account-movement-dependency}</version>
</dependency>

<properties>
    <version.person-client-account-movement-dependency>2.0.2-RELEASE</version.person-client-account-movement-dependency>
